Min-Max Sum - C

#include <stdio.h>
#include<limits.h>
void minMaxSum(int arr[], int size){
    long long total = 0;
    int min = INT_MAX;
    int max = INT_MIN;
    for(int i=0;i<size;i++){
        total +=arr[i];
        if(arr[i]<min){
            min=arr[i];
            
        }
        if(arr[i]>max){
            max=arr[i];
            
        }
    }
    long long minSum = total-max;
    long long maxSum = total - min;
    printf("%lld %lld\n", minSum, maxSum);
    
}
int main(){
    int arr[5];
    for(int i=0; i<5; i++){
        scanf("%d",&arr[i]);
        
    }
    minMaxSum(arr,5);
    return 0;
}
