Pangrams - C

#include<stdio.h>
#include<string.h>
#include<ctype.h>
#define ALBHABET_COUNT 26
char* pangrams(char* s){
    int letters[ALBHABET_COUNT] = {0};
    int i, index;
    int length = strlen(s);
    int unique_count = 0;
    for(i=0; i<length; i++){
        if(isalpha(s[i])){
            index = tolower(s[i])-'a';
            if(letters[index]==0){
                letters[index]= 1;
                unique_count++;
                }
            }
        }
        if(unique_count ==ALBHABET_COUNT)
        return "pangram";
        else
         return "not pangram";
    }
    int main(){
        char s[1000];
        fgets(s, sizeof(s), stdin);
        size_t len = strlen(s);
        if(len>0 &&s[len-1] == '\n'){
            s[len-1] = '\0';
            
        }
        printf("%s\n", pangrams(s));
        return 0;
    }
