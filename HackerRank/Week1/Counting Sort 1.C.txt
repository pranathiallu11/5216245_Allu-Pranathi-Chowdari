Counting Sort 1 - C

#include<stdio.h>
#include<stdlib.h>
int* countingSort(int arr[], int n){
    int *frequency = (int *)calloc(100, sizeof(int));
    for (int i=0; i<n; i++){
        frequency[arr[i]]++;
    }
    return frequency;
}
int main(){
    int n;
    scanf("%d", &n);
    int arr[n];
    for (int i=0; i<n; i++){
        scanf("%d", &arr[i]);
    }
    int* result = countingSort(arr, n);
    for(int i=0; i<100; i++){
        if(i!=0){
            printf(" ");
            }
        printf("%d", result[i]);
}
printf("\n");

    free(result);
    return 0;
    }
